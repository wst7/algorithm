## 题目
https://leetcode-cn.com/problems/qing-wa-tiao-tai-jie-wen-ti-lcof/

## 思路

```
f(n) = f(n - 1) + f(n - 2)
```

```js
/**
 * @param {number} n
 * @return {number}
 */
var numWays = function(n) {
    // f(n) = f(n - 1) + f(n - 2)
    const dp = []
    dp[0] = 1
    dp[1] = 1


    for (i = 2; i <= n; i++) {
        dp[i] = (dp[i - 1] + dp[i - 2]) % 1000000007
    }

    return dp[n]

};

```
