/**
 * @param {number[]} nums
 * @return {number}
 */
var findRepeatNumber = function(nums) {
    const numberMap = {}
    for (let i = 0; i < nums.length; i++) {
        if (numberMap[nums[i]]) {
            return nums[i]
        }
        
        numberMap[nums[i]] = 1
    }
};
